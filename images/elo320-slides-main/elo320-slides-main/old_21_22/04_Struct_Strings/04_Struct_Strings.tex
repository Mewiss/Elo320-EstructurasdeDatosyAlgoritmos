\section{Struct: Agrupando información.}
\begin{frame}[fragile]{Estructuras}
\begin{itemize}
	\item Conjunto heterogéneo de elementos de datos.
	\item Variable: campo, miembro o atributo.
	\item Base de los Tipos de Datos Abstractos (TDA).
	\item Encapsulamiento de información.
	\item Acceso: Operador punto (\texttt{.}).
\end{itemize}
\begin{exampleblock}{C}
\begin{lstlisting}[language=C, basicstyle=\tiny, escapeinside={|}{|}]
struct empleado {
    char nombre[40];
    float sueldo;
};

int main(){
	struct empleado e1;
	e1.sueldo = 550000.00; //acceso a sueldo
	strcpy(e1.nombre, "Juan"); //acceso a nombre
	printf("El empleado |\%|s gana |\%|.2f USD al año", e1.nombre, e1.sueldo); 
	return 0;
}
\end{lstlisting}
\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Estructuras y punteros.}
\begin{itemize}
	\item struct $\rightarrow$ nuevo tipo de datos $\rightarrow$ punteros.
	\item El operador punto se combina con derreferencia, e.g., \texttt{(*var).nombre};
	\item Ojo, \texttt{(*var).nombre $\neq$ *var.nombre.}
	\item En vez, se puede usar el operador \verb+->+. 
\end{itemize}
\begin{exampleblock}{C}
\begin{lstlisting}[language=C, basicstyle=\tiny, escapeinside={|}{|}]
struct empleado {
    char nombre[40];
    float sueldo;
};

int main(){
	struct empleado e1;
	struct empleado *ptr;
	ptr = &e1;
	ptr->sueldo = 550000.00; //acceso a sueldo
	strcpy(ptr->nombre, "Juan"); //acceso a nombre
	printf("El empleado |\%|s gana |\%|.2f USD al año", ptr->nombre, ptr->sueldo); 
	return 0;
}
\end{lstlisting}
\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Definiendo nuevos tipos de datos}
\begin{itemize}
	\item \textbf{typedef} $\rightarrow$ crear nuevos tipos de datos. 
	\item Se utiliza para encapsular structs. 
	\item También para no trabajar con tipos de datos comunes de forma directa.
\end{itemize}
\begin{exampleblock}{C}
\begin{lstlisting}[language=C, basicstyle=\tiny, escapeinside={|}{|}]
typedef float real;

typedef struct empleado {
    char nombre[40];
    real sueldo;
} empleado_t;

int main(){
	empleado_t e1;
	e1.sueldo = 550000.00; //acceso a sueldo
	strcpy(e1.nombre, "Juan"); //acceso a nombre
	printf("El empleado |\%|s gana |\%|.2f USD al año", e1.nombre, e1.sueldo); 
	return 0;
}
\end{lstlisting}
\end{exampleblock}
\end{frame}


\section{Strings}
\begin{frame}[fragile]{Strings}
\begin{itemize}
	\item Estructura de Datos para poder representar palabras.
	\item Trabajan sobre arreglos de caracteres.
	\item La norma es que finalizen con el caracter fin de string: '\textbackslash 0'.
	\item \textbf{string.h} $\rightarrow$ funciones.
	\item Dependen de la existencia del caracter '\textbackslash 0'.
	\item Indexados como todo arreglo.
	\item Comillas (") $\rightarrow$ string, apóstrofe (') $\rightarrow$ char.
 	\item TDA String: funciones + arreglo de caracteres + '\textbackslash 0'.
\end{itemize}

\end{frame}

\begin{frame}[fragile]{Definición}
\begin{exampleblock}{C}
\begin{lstlisting}[language=C, basicstyle=\tiny, escapeinside={|}{|}]
int main(){

	char a[10]; //Tamaño fijo.
	char *b; // Tamaño dinámico.


	a="Hola"; // inválido
	b="Hola"; // válido

	return 0;
}

\end{lstlisting}
\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Inicialización}
\begin{exampleblock}{C}
\begin{lstlisting}[language=C, basicstyle=\tiny, escapeinside={|}{|}]
int main(){

	char a[10]="Hola"; //Tamaño fijo con valor inicial.
	char a2[10]={'H','o','l','a','\0'}; //obligación el '\0'

	char b[5]="Juan"; // Éstas dos son equivalentes.
	char b2[]="Juan" // Tamaño fijo definido con valor inicial.
	
	char *c = "ELO320"; // String constante, tamaño variable. 

	char *d = malloc(...); // String dinámico, puntero.
	strcpy(d, "String Pro");
	free(d); //próx. clase. 	

	a[0]='D'; //válido
	b[0]='D'; //válido
	c[0]='D'; //no valido, segmentation fault.
	d[0]='D'; //valido, por algo es pro.

	return 0;
}

\end{lstlisting}
\end{exampleblock}
\end{frame}

\begin{frame}[fragile]{Biblioteca string.h}
\begin{itemize}
	\item \verb+#include<string.h>+
	\item Contiene funciones y elementos útiles para manejar strings.
\end{itemize}

\begin{exampleblock}{C}
\begin{lstlisting}[language=C, basicstyle=\tiny, escapeinside={|}{|}]
int main(){

	char a[10];
	char *b, *c;
	int tam;

	strcpy(a,"Hola"); // strcpy(a,b) copia en a lo de b
	//strncpy(a,b,n) copia los n primeros caracteres.
	//retornan un puntero a "a". 

	b=" Mundo";
	
	c=strcat(a,b); // concatena a con b;
	//strncat concatena a "a" los primeros n caracteres de b.
	//retornan un puntero a "a". 
	
	tam=strlen(a); //largo de a. no cuenta '\0'.
	// retorna size_t, es un typedef de unsigned long int.

	printf("tamaño: |\%|d\n", tam);

	return 0;
}

\end{lstlisting}
\end{exampleblock}
\end{frame}
