\section{Tipos de Ordenamiento}
\begin{frame}[fragile]
 \frametitle{Ordenamiento por comparación}
\small InsertSort, MergeSort, QuickSort son algoritmos de ordenamiento basados en comparaciones, vale decir, en el contraste de dos valores.

\begin{itemize}
\item Estos algoritmos pueden ser analizados en un arbol binario de decisión, con decisiones $\leq$ y $>$. 
\item El largo de estos árboles determina una complejidad en función de $\log n$. 
\item Además, hay al menos $n$ decisiones por los elementos del árbol. 
\item Por lo tanto, este árbol de decisión actua como conta inferior de todo Ordenamiento por Comparación.
\item Ordenamiento por comparación $\rightarrow \Omega(n \log n)$. 
\end{itemize}

\end{frame}

\begin{frame}[fragile]
 \frametitle{Ordenamiento en tiempo lineal}
\small Existen algoritmos de ordenamiento que no depende de comparaciones, y en este curso veremos dos que mantienen complejidad en tiempo lineal:
\begin{itemize}
\item Counting Sort.
\item Radix Sort.
\end{itemize}

Ambos son de complejidad lineal, vale decir, $\mathcal{O}(n)$. Sin embargo, se es más fino respecto a cada uno de ellos.\\

Además, ambos presentan ciertas limitantes para poder ser aplicados.
\end{frame}

\section{Counting Sort}
\begin{frame}[fragile]
 \frametitle{Counting Sort}
\small Algoritmo de Ordenamiento basado en contar la cantidad de ocurrencias de los elementos en un arreglo. 

\begin{itemize}
\item Complejidad: $\mathcal{O}(n+k)$, con $n$ numero de elementos del arreglo y $k$ el máximo elemento del arreglo.
\item Uso: enteros y representaciones discretas. 
\item No se recomienda para strings, floats, etc. 
\item Es estable.
\item Utiliza bastante memoria extra. 
\end{itemize}
\end{frame}

\begin{frame}[fragile]
 \frametitle{Counting Sort}
\begin{exampleblock}{Counting Sort}
\begin{lstlisting}[language=C, basicstyle=\tiny, escapeinside={|}{|}]
int * counting_sort(int A[], int max){
	int i, ind_sgte, veces;
	int largo = len(A); //funcion largo. ud la crean
	int B[largo]; //arreglo de salida. 
	int C[max+1]; //arreglo contador

	for(i=0;i<=max;i++) //init arreglo
		C[i]=0;
 
	for(i=0;i<largo;i++)
		C[A[i]]++; //contando item de A es índice en C
	
	ind_sgte = 0;
	for(i=0;i<max+1;i++){ //indice siguiente
		veces=C[i];
		C[i] = ind_sgte;		
		ind_sgte += veces;
	}

	for(i=0;i<largo;i++){ //paso a B
		B[C[A[i]]] = A[i];
		C[A[i]]++;
	}
	return B;
}
\end{lstlisting}
\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
 \frametitle{Radix Sort}
\small Algoritmo de Ordenamiento basado en dígitos, para poder ordenar desde las unidades hacia adelante, los elementos en un arreglo. 

\begin{itemize}
\item Complejidad: $\mathcal{O}(d \times (n+k))$, con $d$ la cantidad de digitos usados, $n$ numero de elementos del arreglo y $k$ el máximo elemento del arreglo.
\item Suele ser usado para enteros, y representaciones discretas. ($k=9=$cte.)
\item Por lo tanto, se suele decir que es $\mathcal{O}(d \times n)$. 
\item Se basa en una algoritmo estable para cada etapa $\rightarrow$ Counting Sort. 
\item Utiliza bastante memoria extra. 
\end{itemize}
\end{frame}

 \frametitle{Radix Sort}
\begin{exampleblock}{Radix Sort}
\begin{lstlisting}[language=C, basicstyle=\tiny, escapeinside={|}{|}]
int maximo(int *A, int tam) {
	int i;
	int mayor = A[0];
	for (i = 1; i < tam; i++)
		if (A[i] > mayor)
			mayor = A[i];
	return mayor;
}

void RadixSort(int *A, int tam) {
	int pos;
	int max = maximo(A, tam);

	for (pos=0 ; (max/ pow(10,pos)) > 0; pos++)
		CountingSort(A, tam, pow(10,pos));
}

\end{lstlisting}
\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
 \frametitle{Radix Sort}
\begin{exampleblock}{Radix Sort: built-in Counting Sort}
\begin{lstlisting}[language=C, basicstyle=\tiny, escapeinside={|}{|}]
void CountingSort(int *A, int tam, int lugar){
	int B[tam], i, ind_sgte, veces;
	int max = (A[0] / lugar) |\%| 10; //por si no hay digitos finales
	for (i = 1; i < tam; i++) {
		if (((A[i] / lugar) |\%| 10) > max)
			max = (A[i]/lugar)|\%| 10;
	}
	int C[max + 1];
	for (int i = 0; i < max; ++i)
		C[i] = 0;

	for (int i = 0; i < tam; i++)
		C[(A[i] / lugar) |\%| 10]++;
  
	ind_sgte = 0;
	for(i=0;i<tam;i++){ //indice siguiente
		veces=C[i];
		C[i] = ind_sgte;		
		ind_sgte += veces;
	}

	for(i=0;i<tam;i++){ //paso a B
		B[C[(A[i]/lugar) |\%| 10]] = A[i];
		C[(A[i]/lugar) |\%| 10]++;
	}
}
\end{lstlisting}
\end{exampleblock}
\end{frame}
