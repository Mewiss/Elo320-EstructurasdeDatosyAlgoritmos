Struct en C

  • Un struct permite agrupar variables bajo un mismo nombre.
  • Se accede a los miembros usando el operador . (para variables) o -> (para punteros).
#include <stdio.h>
#include <string.h>

struct Estudiante {
    char nombre[50];
    char apellido[50];
    float nota;};

int main() {
    struct Estudiante e1;
    strcpy(e1.nombre, "Maria");
    strcpy(e1.apellido, "Gonzalez");
    e1.nota = 7.0;
    printf("%s %s - %.2f\n", e1.nombre, e1.apellido, e1.nota);
    return 0;}

typedef struct

Con typedef creamos un alias y evitamos escribir struct al declarar variables. 

#include <stdio.h>
#include <string.h>

typedef struct {
    char nombre[50];
    char apellido[50];
    float nota;
} Estudiante;

int main() {
    Estudiante e1; // no necesitamos escribir 'struct'
    strcpy(e1.nombre, "Pedro");
    strcpy(e1.apellido, "Perez");
    e1.nota = 8.5;
    printf("%s %s - %.2f\n", e1.nombre, e1.apellido, e1.nota);
    return 0;}

Struct y punteros


Estudiante *ptr = malloc(sizeof(Estudiante));
strcpy(ptr->nombre, "Ana");
ptr->nota = 9.0;

printf("%s - %.2f\n", ptr->nombre, ptr->nota);
free(ptr);

Con typedef, el uso de punteros es más legible y no necesitamos escribir struct.

Paso por valor vs paso por referencia con structs


typedef struct {
    char nombre[50];
    char apellido[50];
    float nota;
} Estudiante;

void modificar_por_valor(Estudiante e) {
    e.nota = 7.0; // solo copia
}

void modificar_por_referencia(Estudiante *e) {
    e->nota = 7.0; // modifica original
}

Argumentos por línea de comandos

CLI Arguments: argv y argc

argc y argv: parámetros de ejecución

  • En C, podemos entregar argumentos a los ejecutables mediante la línea de comando.
  • Esta función main contiene dos argumentos: argc y argv.

#include<stdio.h>

int main(int argc, char **argv){ //o *argv[]
    return 0;
}
  • argc: Argument Count → cantidad de parámetros.
  • argv: Argument Vector → los parámetros (strings).

argc y argv: parámetros de ejecución

#include<stdio.h>

int main(int argc, char **argv){ //o *argv[]

    if(argc!=3){
        printf("Debe ingresar 3 parámetros, intentelo nuevamente");
        return -1; // exit(-1) si terminar proceso.
    }

    printf("Argumento 0: %s\n", argv[0]);
    printf("Argumento 1: %s\n", argv[1]);
    printf("Argumento 2: %s\n", argv[2]);

    return 0;
}

argc y argv: transformando tipos

  • atoi() transforma string a int
  • atol() transforma string a long
  • atoll() transforma string a long long
  • atof() transforma string a float.
  • sscanf() lee un string y lo usa como scanf().
  • strtol(): string a long con base (10, 16, 2, ...)
  • strtod(): string a double
  • ⚠️ deben verificar el string!

argc y argv: transformando tipos


#include<stdio.h>
#include<stdlib.h>

int main(int argc, char **argv){ //o *argv[]

    if(argc!=4){
        printf("Debe ingresar 4 parámetros, intentelo nuevamente");
        return -1;
    }

    int a;
    sscanf(argv[2], "%d", &a);

    printf("%d %d %ld\n", atoi(argv[1]), a, strtol(argv[3], NULL, 10));

    return 0;
}

Streams

Streams: Archivos

  • C trabaja con streams para interactuar con dispositivos físicos o virtuales.
  • Este puntero se mueve a través del stream.
  • Se crean streams a archivos con punteros FILE.

#include<stdio.h>

int main(int argc, char **argv){
    FILE *arch; // puntero
    arch = fopen("hola.txt","r"); // apertura stream
    fclose(arch); // cerrar stream
    return 0;}

Streams: modos

  • "r": Lectura (archivo debe existir)
  • "w": Escritura (sobrescribe)
  • "a": Anexar (crea si no existe)
  • "r+": Lectura y escritura (inicio)
  • "w+": Lectura y escritura
  • "a+": Lectura y escritura (apéndice)

Archivos - Escritura

  • fprintf(): como printf, pero con archivo.
  • fputs(): escribe string.
  • fputc(): escribe caracter.

#include<stdio.h>

int main(int argc, char **argv){

    FILE *arch;
    char *saludo = "Hola Mundo";

    arch = fopen("test.txt","w");
    fprintf(arch, "%s\n", saludo);
    fputs("I'll be back", arch);
    fputc('X', arch);
    fclose(arch);
    return 0;
}

Archivos - Lectura

  • fscanf(): como scanf, pero con archivo.
  • fgets(): lee string hasta \n o EOF.
  • fgetc(): lee un caracter.

#include<stdio.h>

int main(int argc, char **argv){
    FILE *arch;
    char saludo[20];
    arch = fopen("test.txt","r");

    while(!feof(arch)){
        fscanf(arch,"%s", saludo);
        fprintf(stderr,"linea: %s\n", saludo);    }

    fclose(arch);
    return 0;
}

Archivos - Lectura


#include<stdio.h>

int main(int argc, char **argv){

    FILE *arch;
    char saludo[5];

    arch = fopen("test.txt","r");

    while(fgets(saludo, 5, arch) != NULL){
        fprintf(stderr,"linea: %s\n", saludo);
    }

    fclose(arch);
    return 0;
}

Streams: Lectura


#include<stdio.h>

int main(int argc, char **argv){

    FILE *arch;
    char saludo;

    arch = fopen("test.txt","r");

    while((saludo=fgetc(arch)) != EOF){
        fprintf(stderr,"linea: %c\n", saludo);
    }

    fclose(arch);
    return 0;
}

Streams: Manejo del puntero

A diferencia de Python, en C se puede manejar el puntero del archivo:

  • rewind(): inicio
  • fseek(): mueve n bytes
  • ftell(): posición actual
  • fgetpos(): guarda posición en fpos_t
  • fsetpos(): mueve a posición guardada

Otros Streams

Existen variados streams en C:

  • Standard I/O: scanf(), printf()
  • Strings: sscanf(), sprintf()
  • Archivos: fscanf(), fprintf()
